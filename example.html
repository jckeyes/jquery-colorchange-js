Set the starting color

$('.color-1').colorChange({r:75, g:255, b:12});

Set the speed

$('.color-2').colorChange({interval: 20});

Set the variance

$('.color-3').colorChange({maxModifier: 100});

Set the range

$('.color-4').colorChange({maxValue: 120, minValue: 100});

Set them all

$('.color-5').colorChange({r:255, g:255, b:12, interval: 75, maxModifier: 50, maxValue: 200, minValue: 100});
